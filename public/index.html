async function sendMessage() { const message = messageInput.value.trim(); if (!message) { showStatus('Введите текст', 'error'); return; } showStatus('Отправка...'); try { const response = await fetch('/api/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ message: message }) }); const data = await response.json(); if (response.ok) { messageInput.value = ''; showStatus('✓ Сообщение отправлено', 'success'); setTimeout(() => statusDiv.style.display = 'none', 2000); } else { showStatus('Ошибка: ' + (data.error || 'неизвестная ошибка'), 'error'); } } catch (error) { showStatus('Ошибка соединения: ' + error.message, 'error'); } }
